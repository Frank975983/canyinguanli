        function getCurrentProducts() {

                        map[p.id].sales += p.sales;

        function calcBom() {

            let total = 0;

            document.querySelectorAll('#bomBody tr').forEach(tr => {

                const mat = materialLibrary.find(m => m.id === tr.querySelector('.bom-sel').value);

                const qty = parseFloat(tr.querySelector('.bom-qty').value) || 0;

                let cost = 0;

                if (mat && qty > 0) {

                    cost = mat.unit === '克' ? (qty / (1 - mat.lossRate)) * (mat.unitPrice / 500) : qty * mat.unitPrice;

                    tr.querySelector('.bom-cost').textContent = `¥${cost.toFixed(2)}`;

                } else { tr.querySelector('.bom-cost').textContent = '¥0.00'; }

                total += cost;

            });

            document.getElementById('totalCost').textContent = `¥ ${total.toFixed(2)}`;

            const price = parseFloat(document.getElementById('prodPrice').value) || 0;

            const gp = price > 0 && total > 0 ? ((price - total) / price * 100).toFixed(1) : 0;

            document.getElementById('grossProfit').textContent = `${gp}%`;

        }

            `;

            modalOverlay.classList.add('show');

        }

        document.getElementById('modalClose').addEventListener('click', closeModal);

        document.getElementById('modalCancel').addEventListener('click', closeModal);

        modalOverlay.addEventListener('click', (e) => { if(e.target === modalOverlay) closeModal(); });

 

        function closeModal() { modalOverlay.classList.remove('show'); currentModalType = ''; editingId = null; }

        function showToast(msg, err = false) { toastMsg.textContent = msg; toast.className = err ? 'toast show error' : 'toast show'; setTimeout(() => toast.className = 'toast', 2500); }

 

        modalSubmit.addEventListener('click', () => {

            if (currentModalType === 'salesRecord') {

                const pid = document.getElementById('recProd').value;

                const qty = parseInt(document.getElementById('recQty').value);

                if (!qty) return showToast('请输入数量', true);

                

                // 找到对应的门店并增加销量

                if (currentStoreId !== 'all') {

                    const p = mockData[currentStoreId].products.find(x => x.id === pid);

                    if(p) p.sales += qty;

                } else {

                    // 总部视角下，默认加到第一个有该菜品的门店

                    for(let sid in mockData) {

                        const p = mockData[sid].products.find(x => x.id === pid);

                        if(p) { p.sales += qty; break; }

                    }

                }

                renderSalesPage(); showToast('销量录入成功');

            }

            else if (currentModalType === 'product') {

                const name = document.getElementById('prodName').value;

                const price = parseFloat(document.getElementById('prodPrice').value);

                if (!name || !price) return showToast('完善信息', true);

                

                const cost = parseFloat(document.getElementById('totalCost').textContent.replace(/[¥,]/g, ''));

                mockData[currentStoreId].products.unshift({ id: 'p'+Date.now(), name, category: document.getElementById('prodCategory').value, price, cost, sales: 0, status: 'active', img: PLACEHOLDER_IMG });

                renderMenuPage(); showToast('菜品保存成功');

            }

            else if (currentModalType === 'material') {

                const name = document.getElementById('matName').value;

                const price = parseFloat(document.getElementById('matPrice').value);

                const loss = parseFloat(document.getElementById('matLoss').value) / 100 || 0;

                if (!name || isNaN(price)) return showToast('完善信息', true);

                

                if (editingId) {

                    const m = materialLibrary.find(x => x.id === editingId);

                    if(m) { m.name = name; m.unit = document.getElementById('matUnit').value; m.unitPrice = price; m.lossRate = loss; }

                } else {

                    materialLibrary.push({ id: 'm'+Date.now(), name, unit: document.getElementById('matUnit').value, unitPrice: price, lossRate: loss });

                }

                renderMaterialPage(); showToast('原料保存成功');

            }

            else if (currentModalType === 'store') {

                const name = document.getElementById('stName').value;

                const manager = document.getElementById('stManager').value;

                const address = document.getElementById('stAddress').value;

                const status = document.getElementById('stStatus').value;

                if (!name || !manager) return showToast('请完善门店信息', true);

 

                if (editingId) {

                    const s = storesMeta.find(x => x.id === editingId);

                    if(s) { s.name = name; s.manager = manager; s.address = address; s.status = status; }

                    showToast('门店信息已更新');

                } else {

                    const newId = 'store_' + Date.now();

                    storesMeta.push({ id: newId, name, manager, address, status });

                    mockData[newId] = { products: [], revenue: 0, expense: 0 };

                    showToast('新门店添加成功');

                }

                renderStoreDropdown();

                refreshData();

            }

            closeModal();

        });

            renderStoreDropdown();

            refreshData();

            initSalesChart();

        });
